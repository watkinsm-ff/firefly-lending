// ---- PAYMENT CALCULATOR ----

function fmt(n) {

  return '$' + Math.round(n).toLocaleString('en-US');

}


function calcRate(amt, term) {

  // Equal 50/50 weight between term and amount

  // Short term (6mo) and small amount ($5K) both push toward 12%

  // Long term (24mo) and large amount ($42K) both push toward 8%

  const termPremium  = ((24 - term) / 18) * 2;          // 0–2%

  const amtPremium   = ((42000 - amt) / 37000) * 2;     // 0–2%

  const raw = 8 + termPremium + amtPremium;

  return Math.min(12, Math.max(8, Math.round(raw * 4) / 4)); // round to nearest 0.25, cap 8-12

}


function updatePreview() {

  const amt = parseFloat(document.getElementById('loanAmt').value);

  const term = parseInt(document.getElementById('loanTerm').value);

  const ptype = document.querySelector('input[name="ptype"]:checked').value;


  const rate = calcRate(amt, term);


  document.getElementById('amtVal').textContent = fmt(amt);

  document.getElementById('rateVal').textContent = rate.toFixed(2) + '%';

  document.getElementById('termVal').textContent = term + ' mo';


  // Update the visual rate track

  const pct = ((rate - 8) / 4) * 100;

  const fill = document.getElementById('rateTrackFill');

  const thumb = document.getElementById('rateThumb');

  if (fill) fill.style.width = pct + '%';

  if (thumb) thumb.style.left = pct + '%';


  const monthlyRate = rate / 100 / 12;

  const points = amt * 0.02;

  let monthly, totalInterest, balloon;


  if (ptype === 'pni') {

    // Standard amortizing

    monthly = amt * (monthlyRate * Math.pow(1 + monthlyRate, term)) / (Math.pow(1 + monthlyRate, term) - 1);

    totalInterest = (monthly * term) - amt;

    balloon = 0;

    document.getElementById('prev_balloon_label').textContent = 'Balloon at Maturity';

    document.getElementById('prev_balloon').textContent = 'None';

  } else {

    // Interest only + balloon

    monthly = amt * monthlyRate;

    totalInterest = monthly * term;

    balloon = amt;

    document.getElementById('prev_balloon_label').textContent = 'Balloon at Maturity';

    document.getElementById('prev_balloon').textContent = fmt(balloon);

  }


  const totalCost = totalInterest + points;


  document.getElementById('prev_monthly').textContent = fmt(monthly) + '/mo';

  document.getElementById('prev_interest').textContent = fmt(totalInterest);

  document.getElementById('prev_points').textContent = fmt(points);

  document.getElementById('prev_total').textContent = fmt(totalCost);


  // Prepayment warning

  const threshold = Math.floor(term * 0.75);

  const penaltyMonths = threshold;

  const penaltyInterest = ptype === 'pni'

    ? (monthly * penaltyMonths - amt * (1 - Math.pow(1 + monthlyRate, -term) / Math.pow(1 + monthlyRate, penaltyMonths - term)))

    : monthly * penaltyMonths;

  const penaltyAmt = amt * 0.02;


  document.getElementById('prepayWarning').innerHTML =

    `<strong>Prepayment Note:</strong> If this loan is paid off before month ${threshold} of ${term} (75% of term), a prepayment penalty applies. Penalty is calculated as interest due through month ${threshold}, plus an additional 2% of outstanding principal. Early payoff does not reduce the lender's return below minimum yield expectations.`;

}
